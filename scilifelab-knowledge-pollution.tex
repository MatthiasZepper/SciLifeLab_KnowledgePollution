		\begin{alertblock}{Aggravation}
			\begin{itemize}

\begin{frame}{LLM slop: An issue of incentives and objectives}
	%\metroset{block=fill}
	\begin{block}{Incentives stemming from LLM training evaluation}
		\begin{itemize}  
			\item Hallucinations in LLMs are inevitable with the current model training approach and its main objectives.\footnote{For example, most industry benchmarks use \emph{0-1 scoring}, where a correct answer earns 1 point and both a wrong answer and abstention earn 0 points} \citeme{kalai2024,kalai2025}.
		\end{itemize}
	\end{block}
	
	\begin{block}{Incentives in human psychology}
		\begin{itemize}
			\item Overconfidence enhances a person's social status, and allows them to exert influence over groups despite poor performance\citeme{anderson2012, ronay2019}.
			\item The brain treats confidence expressed by others as information about expected reward or success; a strong signal in the ventromedial prefrontal cortex (vmPFC) is recorded in fMRI studies\citeme{campbell-meiklejohn2016, molenberghs2016}.
			
		\end{itemize}
	\end{block}
\end{frame}

\section{Summary}

% % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % % 


\begin{frame}{Is AI a new threat to science?}
	\begin{block}{Scientific process}
		\begin{itemize}
			\item Expert human curation capacity is scare. Generative AI strains peer review with slop and conclusive fraud. \remark{Improved detection tools? } 
			\item AI enables powerful new research tools, e.g. for data analysis.
			\remark{Criteria for assessing academic merits in the age of AI-Ghostwriters}   
		\end{itemize}
	\end{block}
	\begin{block}{Scientific results}
		\begin{itemize}
			\item The provisional, revisable nature of findings is known to scientists.
			\remark{Findable, Accessible, Interoperable, and Reusable (FAIR) data}  
		\end{itemize}
	\end{block}
	\begin{block}{Science-informed policy}
		\begin{itemize}
			\item AI-enabled production of low-quality or misleading research for political or economic gains meets weakened institutions. 
			\remark{Social media speed + polarization, anti-elitist distrust of experts}   
		\end{itemize}
	\end{block}
\end{frame}

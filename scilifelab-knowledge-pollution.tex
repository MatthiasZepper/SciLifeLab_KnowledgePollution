\begin{frame}{Knowledge pollution by vandalized Wikipedia article}

		\item Major outlets like \href{https://www.spiegel.de/politik/deutschland/in-eigener-sache-falscher-wilhelm-bei-minister-guttenberg-a-606912.html}{Spiegel} or \href{https://taz.de/Wikipedia-Manipulation/!5168061/}{TAZ} publish corrections in the following days.
	\end{itemize}
\end{frame}

\begin{frame}{Lessons learned}
	\metroset{block=fill}
		\begin{alertblock}{Aggravation}
			\begin{itemize}
				\item The appointment was decided Sunday night, when editorial offices were short-staffed. High time pressure.
				\item Once the polluted knowledge had spread widely to creditable outlets, correcting the error became difficult.
			\end{itemize}
		\end{alertblock}

		\begin{block}{Mitigation}
			\begin{itemize}
				\item Atomic issue; the minister himself could be asked.
				\item Wikipedia has a full audit trail. It is retraceable who edited what and when even after 17 years.
			\end{itemize}
		\end{block}
\end{frame}
